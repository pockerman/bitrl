117  _LunarLanderEnv<TimeStepType, SpaceType>::make(const std::string& version,
118  const std::unordered_map<std::string, std::any>& options)
119  {
120  if(this->is_created()){

123 
124  nlohmann::json ops;
125 
126  if (auto gravity_itr = options.find("gravity"); gravity_itr != options.end())

129  ops["gravity"] = gravity;

131

133  {
134  auto wind = std::any_cast<bool>(enable_wind_itr->second);
135  ops["enable_wind"] = wind;
136  }
137 
138  if (auto wind_power_itr = options.find("wind_power"); wind_power_itr != options.end())
139  {
140  auto wind_power = std::any_cast<real_t>(wind_power_itr->second);
141  ops["wind_power"] = wind_power;
142  }
143 
144  if (auto turbulence_power_itr = options.find("turbulence_power"); turbulence_power_itr != options.end())
145  {
146  auto turbulence_power = std::any_cast<real_t>(turbulence_power_itr->second);
147  ops["turbulence_power"] = turbulence_power;
148  }
149 
150  auto response = this -> get_api_server().make(this -> env_name(),
151  this -> cidx(),
152  version, ops);
153 
154  this -> set_version_(version);
155  this -> set_make_options_(options);
156  this -> make_created_();
