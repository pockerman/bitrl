196  // loop over the sensor topics and perform

215  void
216  EKFSensorFusion<MotionModelType, ObservationModelType>::predict(real_t dt,
217  const std::string& input_message,
218  const std::string& sensor_message,
219  const topic_type& topic)
220  {
223  motion_model_->evaluate(dt, input_message, sensor_message, topic);
224 
225  auto& P = (*this)["P"];
226  auto& Q = (*this)["Q"];
227 
228  // get the Jacobian description for the
229  // motion model
230  auto& F = motion_model_->get_matrix("F");
231  auto F_T = F.transpose();
232 
233  P = F * P * F_T;
234 
235  if (motion_model_ -> has_matrix("L"))
236  {
237  auto& L = motion_model_->get_matrix("L");
238  auto L_T = L.transpose();
239 
240  P += L * Q * L_T;
241  }
242  else
243  {
244  // Most EKFs simply add Q
245  P += Q;
246  }
247  }

249  template<typename MotionModelType, typename ObservationModelType>
250  void
251  EKFSensorFusion<MotionModelType, ObservationModelType>::update(real_t dt,
252  const std::string& sensor_message,
253  const topic_type& topic)
254  {
255  auto& state = motion_model_->get_state();
256  auto& P = (*this)["P"];
257  auto& R = (*this)["R"];
258 
259  auto z = observation_model_ -> convert_sensor_message(sensor_message, topic);
260  auto zpred = observation_model_ -> evaluate(state.as_vector());
261 
262  auto& H = observation_model_->get_matrix("H");
263  auto H_T = H.transpose();
264 
265  // compute \partial{h}/\partial{v} the jacobian of the observation model
266  // w.r.t the error vector
267  auto& M = observation_model_->get_matrix("M");
268  auto M_T = M.transpose(); //trans(M);
269 
270  try{
271 
272  // S = H*P*H^T + M*R*M^T
273  auto S = H*P*H_T + M*R*M_T;
274  auto S_inv = S.inverse();
275 
276  if(has_matrix("K")){
277  auto& K = (*this)["K"];
278  K = P*H_T*S_inv;
279  }
280  else{
281  auto K = P*H_T*S_inv;
282  with_matrix("K", K);
283  }
284 
285  auto& K = (*this)["K"];
286  auto innovation = z - zpred;
287 
288  // we need to take the transpose
289  auto innovation_t = innovation.transpose();
290 
291  if(K.cols() != innovation_t.rows()){
292  throw std::runtime_error("Matrix columns: "+
293  std::to_string(K.cols())+
294  " not equal to vector size: "+
295  std::to_string(innovation_t.rows()));
296  }
297 
298  //auto vec = K * innovation_t;
299  state += K * innovation_t;
300 
301  //IdentityMatrix<real_t> I(state.size());
302  auto I = matrix_type::Identity(state.size(), state.size());
304  P = (I - K*H)*P;
305  }
306  catch(...){
307 
308  // this is a singular matrix what
309  // should we do? Simply use the predicted
310  // values and log the fact that there was a singular matrix
311  throw;
312  }
313  }

315 
316  }
